import org.springframework.cache.Cache;
import org.springframework.cache.support.SimpleValueWrapper;
import org.springframework.stereotype.Component;

import java.util.Collection;
import java.util.Map;
import java.util.Objects;
import java.util.concurrent.*;

/**
 * 自定义缓存实现类
 * @Author: CZM
 */
@Component
public class MyCache implements Cache {

    /**
     * 存储缓存数据的Map
     */
    private Map<String, Object> storage = new ConcurrentHashMap<>();

    /**
     * 缓存过期时间的Map
     */
    private Map<String, Long> expiration = new ConcurrentHashMap<>();


    @Override
    public String getName() {
        return null;
    }

    @Override
    public Object getNativeCache() {
        return null;
    }

    /**
     * 判断是否存在指定的键
     *
     * @param key 键
     * @return true 存在 false 不存在
     */
    public boolean hasKey(String key){
        return storage.containsKey(key);
    }

    /**
     * 根据键获取缓存值
     *
     * @param key 键
     * @return 缓存值
     */
    @Override
    public ValueWrapper get(Object key) {
        String k = key.toString();
        if (isExpired(k)) {
            evict(k);
            return null;
        }
        Object value = storage.get(k);
        return Objects.isNull(value) ? null : new SimpleValueWrapper(value);
    }


    @Override
    public <T> T get(Object key, Class<T> type) {
        return null;
    }

    @Override
    public <T> T get(Object key, Callable<T> valueLoader) {
        return null;
    }

    /**
     * 添加缓存数据
     *
     * @param key   键
     * @param value 值
     */
    public void put(Object key, Object value) {
        put(key, value, 0, TimeUnit.SECONDS);
    }

    /**
     * 添加缓存数据和过期时间
     *
     * @param key     键
     * @param value   值
     * @param timeout 过期时间（秒）
     */
    public void put(Object key, Object value, long timeout, TimeUnit timeUnit) {
        if (Objects.isNull(value)) {
            return;
        }
        String k = key.toString();
        storage.put(k, value);
        if (timeout > 0) {
            long expireTime = System.currentTimeMillis() + timeUnit.toMillis(timeout);
            expiration.put(k, expireTime);
        }
    }

    /**
     * 清除指定键的缓存数据
     *
     * @param key 键
     */
    @Override
    public void evict(Object key) {
        String k = key.toString();
        storage.remove(k);
        expiration.remove(k);
    }

    /**
     * 删除集合中的对象对应的缓存数据
     *
     * @param collection 集合
     * @return 是否删除成功
     */
    public boolean deleteObject(final Collection collection){
        collection.forEach(o -> {
            storage.remove(o.toString());
        } );
        return true;
    }

    /**
     * 根据模式获取集合中的键
     *
     * @param pattern 模式
     * @return 键的集合
     */
    public Collection<String> keys(final String pattern){
        return storage.keySet();
    }

    /**
     * 清除全部缓存数据
     */
    @Override
    public void clear() {
        storage.clear();
        expiration.clear();
    }

    /**
     * 判断缓存是否过期
     *
     * @param key 键
     * @return true 过期 false 未过期
     */
    private boolean isExpired(String key) {
        Long expireTime = expiration.get(key);
        return Objects.nonNull(expireTime) && expireTime < System.currentTimeMillis();
    }
}
