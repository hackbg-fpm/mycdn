import com.deepoove.poi.XWPFTemplate;

import javax.servlet.http.HttpServletResponse;

import java.security.MessageDigest;
import java.text.SimpleDateFormat;

/**
 * word 模版处理工具
 */

public class WordUtil {
    /**
     * 本地文件下载
     *
     * @param out:
     * @param targetPath:
     * @return: void
     * @Author: CZM
     */
    public static void downloadLocalFile(OutputStream out, String targetPath) throws IOException {
        // 检查目标文件是否存在
        File file = new File(targetPath);
        if (!file.exists()) {
            throw new FileNotFoundException("File does not exist: " + targetPath);
        }
        // 使用try-with-resources确保InputStream自动关闭
        try (InputStream in = new FileInputStream(targetPath)) {
            byte[] buffer = new byte[1024];
            int len;
            while ((len = in.read(buffer)) > 0) {
                out.write(buffer, 0, len);
            }
            // 输出缓冲区中的剩余数据
            out.flush();
        } // 输入流会在这里自动关闭
    }

    /**
     * 导出Word
     *
     * @param response:
     * @param templatePath:模板地址
     * @param paramMap:         数据
     * @return: void
     * @Author: CZM
     */
    public static void downloadWordResponse(HttpServletResponse response, String templatePath, JSONObject paramMap) throws Exception {
        try {
            // 确保模板路径有效
            if (!Files.exists(Paths.get(templatePath))) {
                throw new Exception("Template file not found: " + templatePath);
            }

            // 计算文件名的哈希值
            MessageDigest messageDigest = MessageDigest.getInstance("MD5");
            messageDigest.update(templatePath.getBytes(StandardCharsets.UTF_8));
            byte[] digest = messageDigest.digest();
            String fileName = Base64.getEncoder().encodeToString(digest);

            // 设置响应头
            response.setContentType("application/vnd.openxmlformats-officedocument.wordprocessingml.document");
            response.setHeader("Content-Disposition", "attachment;filename=\"template.docx\"");
            response.setHeader("Content-Disposition", "attachment;filename*=UTF-8''" + fileName + ".docx");
            response.setCharacterEncoding("UTF-8");

            // 使用XWPFTemplate渲染模板
            XWPFTemplate template = XWPFTemplate.compile(templatePath).render(paramMap);
            try (OutputStream out = response.getOutputStream()) {
                template.write(out);
                out.flush();
            }
        } catch (IOException | TemplateEngineException e) {
            // 可以根据实际情况记录日志
            e.printStackTrace();
        }
    }


    /**
     * 网络下载和本地下载PDF和Word
     *
     * @param response:
     * @param templatePath: 模板文件
     * @param paramMap:     数据
     * @return: void
     * @Author: CZM
     */
    public static void downloadPdfResponse(HttpServletResponse response, String templatePath, JSONObject paramMap) throws Exception {
        String profile = "F:\\我的文件\\";
        // String profile = "/tmp/pdf/";
        String random = UUID.randomUUID().toString().replace("-", "").substring(3);
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
        String format = simpleDateFormat.format(new Date()) + random;
        String savePath = profile + format + ".docx";
        String pdfSavePath = profile + format + ".pdf";

        response.setContentType("application/x-download;charset=utf-8");
        response.setCharacterEncoding("UTF-8");
        response.setHeader("Content-disposition", "attachment;filename=" + System.currentTimeMillis() + ".pdf");

        try (OutputStream out = response.getOutputStream();
             XWPFTemplate template = XWPFTemplate.compile(templatePath).render(paramMap)) {
            File reportFile = new File(savePath);
            // 判断目标文件所在目录是否存在
            if (!reportFile.getParentFile().exists()) {
                // 如果目标文件所在的目录不存在，则创建父目录
                reportFile.getParentFile().mkdirs();
            }
            template.writeToFile(savePath);
            PDFHelper.doc2pdfV2(savePath, pdfSavePath);
            downloadLocalFile(out, pdfSavePath);
        }
    }
}

    public static void pdf2Image(String sourceFile, String targetFile) {
        File file = new File(sourceFile);
        PDDocument pdDocument;
        try {

            pdDocument = PDDocument.load(file);
            PDFRenderer renderer = new PDFRenderer(pdDocument);
            /* dpi越大转换后越清晰，相对转换速度越慢 */
            File dstFile = new File(targetFile);
            BufferedImage image = renderer.renderImageWithDPI(0, 300);
            ImageIO.write(image, "png", dstFile);
            System.out.println("PDF文档转PNG图片成功！");


        } catch (IOException e) {
            e.printStackTrace();
        }
    }
